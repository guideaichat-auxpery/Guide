    function appendBubble(text, cls='ai') {
      const b = document.createElement('div');
      b.className = 'chat-bubble ' + (cls === 'ai' ? 'ai' : 'user');
      b.innerHTML = text;
      // set avatar background for ai bubble if available
      if (cls === 'ai') {
        b.style.setProperty('--ai-avatar-url', "url('/static/assets/maria-montessori-avatar.png')");
        b.style.background = ''; // leave to CSS classes
        // set pseudo-element background via inline style for compatibility
        b.querySelectorAll('::before'); // noop to keep linter happy
        b.style.setProperty('background-image', '');
      }

      chatHistory.scrollTo({ top: chatHistory.scrollHeight, behavior: 'smooth' });

    async function callChatAPI(message, role='educator') {
      try {
        const res = await fetch('/api/chat', {
          method:'POST',
          headers:{ 'Content-Type':'application/json' },
          body: JSON.stringify({ message, role })
        });
        return res.ok ? res.json() : { error: 'Network error' };
      } catch (err) {
        return { error: err.message || String(err) };
      }
    }

    sendBtn.addEventListener('click', async () => {
      const text = userInput.value.trim();
      if (!text) return;
      appendBubble(text, 'user');
      userInput.value = '';
      const typingEl = createTypingElement();
      const typingWrap = document.createElement('div');
      typingWrap.className = 'chat-bubble ai';
      typingWrap.appendChild(typingEl);

      chatHistory.scrollTo({ top: chatHistory.scrollHeight, behavior: 'smooth' });

      const result = await callChatAPI(text, 'student'); // default to 'student' for inquiry style

        let aiHtml = result.response ? result.response.replace(/\n/g, '<br>') : 'No response.';
        appendBubble(aiHtml, 'ai');
      }
    });

      if (e.key === 'Enter' && !e.shiftKey) {
        e.preventDefault();
        sendBtn.click();
      }
    });

# Minimal Flask server to serve this front-end and proxy to OpenAI
from flask import Flask, render_template, request, jsonify
import os, openai
from datetime import datetime
from dotenv import load_dotenv

load_dotenv()
openai.api_key = os.getenv("OPENAI_API_KEY")

app = Flask(__name__, static_folder='static', template_folder='templates')

def inject_curriculum_context(user_input):
    montessori_context = (
        "Dr Maria Montessori emphasized education for peace, moral independence, "
        "and purposeful activity. Adolescents should grapple with ethical and social complexity."
    )
    australian_context = (
        "Australian Curriculum v9: emphasise inquiry, intercultural understanding, ethical reasoning."
    )
    return f"Context:\\n{montessori_context}\\n{australian_context}\\n\\nUser Input:\\n{user_input}"

@app.route("/api/chat", methods=["POST"])
def chat():
    try:
        data = request.get_json()
        user_input = data.get("message","")
        role = data.get("role","student")
        enriched = inject_curriculum_context(user_input)
        style = ("You are a Montessori-aligned guide for adolescents. Offer open-ended provocations, "
                 "ethical questions, and multiple pathways rather than step-by-step solutions.") if role == "student" else (
                 "You are an educator support assistant. Provide scaffolding, rubric links and assessment ideas.")
        # Call OpenAI ChatCompletion (use whichever SDK call your environment expects)
        resp = openai.ChatCompletion.create(
            model="gpt-4o-mini",
            messages=[
                {"role":"system","content":"You are GuideChat, a Montessori-aligned assistant."},
                {"role":"user","content": enriched + "\\n\\n" + style}
            ],
            max_tokens=3000,
            temperature=0.8
        )
        out = resp.choices[0].message['content'].strip()
        # return structured if you wish to highlight curriculum later
        return jsonify({"response": out, "timestamp": datetime.utcnow().isoformat()})
    except Exception as e:
        return jsonify({"error": str(e)}), 500

@app.route("/")
def index():
    return render_template("index.html")

if __name__ == "__main__":
    # Replit likes port 8080; adjust if your environment demands another
    app.run(host="0.0.0.0", port=8080, debug=True)
