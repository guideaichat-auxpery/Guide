Subject Tag Selector (Student Side)

When a student clicks “Start New Chat,” show a lightweight modal or dropdown:

import { useState } from "react";
import { Button } from "@/components/ui/button";
import { Select, SelectTrigger, SelectValue, SelectContent, SelectItem } from "@/components/ui/select";

export default function NewChatModal({ onCreate }) {
  const [subject, setSubject] = useState("");

  const subjects = [
    "Mathematics",
    "Science",
    "Language",
    "Humanities",
    "Cosmic Education",
    "Personal Project",
    "General",
  ];

  return (
    <div className="p-4 space-y-4">
      <h2 className="text-lg font-semibold">Start a New Work Conversation</h2>
      <Select onValueChange={setSubject}>
        <SelectTrigger className="w-full">
          <SelectValue placeholder="Select Subject or Class" />
        </SelectTrigger>
        <SelectContent>
          {subjects.map((s) => (
            <SelectItem key={s} value={s}>{s}</SelectItem>
          ))}
        </SelectContent>
      </Select>

      <Button
        disabled={!subject}
        onClick={() => onCreate(subject)}
        className="w-full mt-2"
      >
        Begin Chat
      </Button>
    </div>
  );
}

Teacher Dashboard Filters

Create a ChatFilterBar and filtered list display.

import { useState } from "react";
import { Input } from "@/components/ui/input";
import { Select, SelectTrigger, SelectValue, SelectContent, SelectItem } from "@/components/ui/select";

export default function ChatFilterBar({ students, subjects, onFilter }) {
  const [student, setStudent] = useState("");
  const [subject, setSubject] = useState("");

  const handleFilter = () => onFilter({ student, subject });

  return (
    <div className="flex flex-wrap gap-2 items-center mb-4">
      <Select onValueChange={setStudent}>
        <SelectTrigger className="w-40"><SelectValue placeholder="Student" /></SelectTrigger>
        <SelectContent>
          {students.map((s) => <SelectItem key={s.id} value={s.id}>{s.name}</SelectItem>)}
        </SelectContent>
      </Select>

      <Select onValueChange={setSubject}>
        <SelectTrigger className="w-40"><SelectValue placeholder="Subject" /></SelectTrigger>
        <SelectContent>
          {subjects.map((s) => <SelectItem key={s} value={s}>{s}</SelectItem>)}
        </SelectContent>
      </Select>

      <Button onClick={handleFilter}>Filter</Button>
    </div>
  );
}

        </div>
      ))}
    </div>
  );
}


Student Dashboard Grouping

Group the student’s chat history by subjectTag:

function StudentChatGroups({ chats }) {
  const grouped = chats.reduce((acc, chat) => {
    if (!acc[chat.subjectTag]) acc[chat.subjectTag] = [];
    acc[chat.subjectTag].push(chat);
    return acc;
  }, {});

  return (
    <div className="space-y-6">
      {Object.entries(grouped).map(([subject, subjectChats]) => (
        <div key={subject}>
          <h3 className="text-lg font-semibold mb-2">{subject} Work</h3>
          <ChatList chats={subjectChats} />
        </div>
      ))}
    </div>
  );
}
