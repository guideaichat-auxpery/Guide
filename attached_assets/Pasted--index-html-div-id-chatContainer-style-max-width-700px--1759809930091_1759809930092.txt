<script type="module">
import Database from "@replit/database";
const db = new Database();

const chatOutput = document.getElementById('chatOutput');
const userInput = document.getElementById('userInput');
const sendBtn = document.getElementById('sendBtn');
const tooltip = document.querySelector('span');

// ---------- Curriculum Keywords ----------
const curriculumKeywords = {
  Geography: ["environment", "urbanization", "sustainability", "place", "space", "interconnectedness"],
  DesignAndTechnologies: ["innovation", "design thinking", "prototype", "technology", "functionality"],
  English: ["persuasive text", "informational text", "communication", "presentation", "literacy"],
  CivicsAndCitizenship: ["ethics", "community", "decision-making", "rights", "responsibilities", "intercultural understanding"]
};

// ---------- Helper Functions ----------
function extractCurriculumKeywords(userInput) {
  const foundKeywords = [];
  for (const [subject, keywords] of Object.entries(curriculumKeywords)) {
    keywords.forEach(keyword => {
      if (new RegExp(`\\b${keyword}\\b`, "i").test(userInput)) {
        foundKeywords.push({subject, keyword});
      }
    });
  }
  return foundKeywords;
}

async function updateTrendingKeywords(foundKeywords) {
  if (!foundKeywords.length) return;
  const trending = (await db.get("trending_keywords")) || {};
  foundKeywords.forEach(({subject, keyword}) => {
    if (!trending[subject]) trending[subject] = {};
    trending[subject][keyword] = (trending[subject][keyword] || 0) + 1;
  });
  await db.set("trending_keywords", trending);
}

// Choose subject color class
function getSubjectClass(subject) {
  switch (subject) {
    case "Geography": return "subject-geography";
    case "DesignAndTechnologies": return "subject-design";
    case "English": return "subject-english";
    case "CivicsAndCitizenship": return "subject-civics";
    default: return "subject-geography";
  }
}

// ---------- Event Listeners ----------
userInput.addEventListener('mouseenter', () => tooltip.style.visibility = 'visible');
userInput.addEventListener('mouseleave', () => tooltip.style.visibility = 'hidden');

sendBtn.addEventListener('click', async () => {
  const message = userInput.value.trim();
  if (!message) return;

  // Display student message
  const studentBubble = document.createElement('div');
  studentBubble.className = 'chat-bubble student-bubble';
  studentBubble.textContent = message;
  chatOutput.appendChild(studentBubble);
  chatOutput.scrollTo({top: chatOutput.scrollHeight, behavior: 'smooth'});
  userInput.value = "";

  // ---------- Keyword Extraction ----------
  const foundKeywords = extractCurriculumKeywords(message);
  const mainSubject = foundKeywords.length ? foundKeywords[0].subject : "Geography";

  // Typing indicator
  const typingBubble = document.createElement('div');
  typingBubble.className = `chat-bubble ai-bubble ${getSubjectClass(mainSubject)}`;
  typingBubble.innerHTML = `
    <div class="typing">
      <div class="typing-dot"></div>
      <div class="typing-dot"></div>
      <div class="typing-dot"></div>
    </div>
  `;
  chatOutput.appendChild(typingBubble);
  chatOutput.scrollTo({top: chatOutput.scrollHeight, behavior: 'smooth'});

  // Wait a bit for realism
  await new Promise(r => setTimeout(r, 1500));

  // Replace typing with AI response
  typingBubble.innerHTML = `
    "Let us not raise our children for today's world. This world will not exist when they grow up."
    <br><br>
    How might this idea challenge your perspective on <strong>${mainSubject}</strong> and its role in shaping ethical, sustainable futures?
  `;
});
</script>
